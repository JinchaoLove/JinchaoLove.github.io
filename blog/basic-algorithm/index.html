def longestPalindrome(s: str) -> str:
    n = len(s)
    if n <= 1:
        return s
    max_s, max_len = "", 0
    for i in range(n):
        if n - 1 - i < (max_len - 1) / 2:
            break  # ÊèêÂâçÁªàÊ≠¢
        # Â§ÑÁêÜÂ•áÊï∞ÈïøÂ∫¶ÁöÑÂõûÊñáÂ≠ê‰∏≤Ôºå‰ª•i‰∏∫‰∏≠ÂøÉÂêë‰∏§ËæπÁßªÂä®
        left, right = i, i
        while True:
            if left < 0 or right > n - 1:
                break
            if s[left] == s[right]:
                left -= 1
                right += 1
            else:
                break  # Ê≥®ÊÑèÊâÄÊúâbreakÁöÑÊÉÖÂÜµ
        cur_len = right - left - 1
        if cur_len > max_len:
            max_s = s[left + 1 : right]
            max_len = cur_len
        # Â§ÑÁêÜÂÅ∂Êï∞ÈïøÂ∫¶ÁöÑÂõûÊñáÂ≠ê‰∏≤

        while True:

                break
            if s[left] == s[right]:
                left -= 1
                right += 1
            else:
                break
        cur_len = right - left - 1
        if cur_len > max_len:
            max_s = s[left + 1 : right]
            max_len = cur_len
    return max_s

def hash_search(arr, query):
    # ÂìàÂ∏åÊü•ÊâæÔºåÁî®‰∫éÊó†Â∫èÊï∞ÁªÑ

def binary_search(array, query):
    """ Two points. [low, high] will be splitted:
        (1) [low, mid - 1]
        (2) [mid]
        (3) [mid + 1, high]
    """
    low, high = 0, len(array) - 1  # Èó≠Âå∫Èó¥ [left, right]
    while low <= high:
        mid = low + (high - low) // 2  # Èò≤Ê∫¢Âá∫
        val = array[mid]
        # array[low:mid], array[mid], array[mid+1:high+1]
        if val == query:
            return mid
        if val < query:
            low = mid + 1
        else:
            high = mid - 1
    return None

def binary_search_recur(array, low, high, query):
    """ Recurrence. [low, high] will be splitted:
        (1) [low, mid - 1]
        (2) [mid]
        (3) [mid + 1, high]
    """
    if low > high:
        return -1
    mid = low + (high - low) // 2   # This mid will not break integer range
    if query < array[mid]:
        return binary_search_recur(array, low, mid - 1, query)  # Go search in the left subarray
    if query > array[mid]:
        return binary_search_recur(array, mid + 1, high, query)  # Go search in the right subarray
    return mid  # `array[mid] = query`, stop recurrence

def search(nums: List[int], target: int) -> int:
    low, high = 0, len(nums) - 1
    while low <= high:
        mid = low + (high - low) // 2
        val = nums[mid]
        # print(mid, nums[low:mid], nums[mid], nums[mid+1:high+1])
        if val == target:
            return mid
        if low < mid and nums[low] <= nums[mid - 1]:
            # Â∑¶ËæπÊúâÂ∫èÔºåÂÖàÂà§Êñ≠ÊòØÂê¶Âú®Â∑¶Ëæπ
            if nums[low] <= target and target <= nums[mid - 1]:
                high = mid - 1
            else:
                low = mid + 1
        elif mid < high:
            # Âè≥ËæπÊúâÂ∫èÔºåÂÖàÂà§Êñ≠ÊòØÂê¶Âú®Âè≥Ëæπ
            if nums[mid + 1] <= target and target <= nums[high]:
                low = mid + 1
            else:
                high = mid - 1
        else:
            return -1
    return -1
